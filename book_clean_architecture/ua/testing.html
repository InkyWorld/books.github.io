Стратегия Тестирования

Как Clean Architecture делает ваш код тестируемым. Разные слои — разные типы тестов.

Пирамида Тестирования

Clean Architecture идеально ложится на классическую пирамиду тестов.

┌─────────────────┐ │ E2E Tests │ ◀── Presentation (API) │ (медленные) │ Требуют: запущенный сервер └────────┬────────┘ │ ┌─────────────┴─────────────┐ │ Integration Tests │ ◀── Infrastructure + Application │ (средняя скорость) │ Требуют: тестовую БД └────────────┬──────────────┘ │ ┌─────────────────────────┴─────────────────────────┐ │ Unit Tests │ ◀── Domain + Application │ (БЫСТРЫЕ!) │ Требуют: НИЧЕГО │ Это основа. 80% всех тестов должны быть здесь. │ └───────────────────────────────────────────────────┘
Unit

Domain Layer

Чистый Python. Без моков. Тестируем Entities и Value Objects.

Integration

Application + Infra

Use Cases с моками репозиториев или реальной тестовой БД.

E2E

Presentation

HTTP-запросы к реальному API. TestClient FastAPI/Django.

1. Тестирование Domain Layer (Unit Tests)

Это самые простые и быстрые тесты. Домен — чистый Python. Никаких зависимостей, никаких моков. Просто создаем объекты и проверяем их поведение.

1.1 Тестирование Value Objects

# tests/domain/test_value_objects.py
import pytest
from domain.value_objects import Money, Email
from domain.exceptions import NegativeMoneyError, InvalidEmailError


class TestMoney:
    """Тесты для Value Object Money"""

    def test_create_valid_money(self):
        """Успешное создание денежной суммы"""
        money = Money(100.50, "USD")
        
        assert money.amount == 100.50
        assert money.currency == "USD"

    def test_create_zero_money(self):
        """Ноль — допустимая сумма"""
        money = Money(0)
        assert money.amount == 0

    def test_negative_money_raises_error(self):
        """ИНВАРИАНТ: Отрицательная сумма запрещена"""
        with pytest.raises(NegativeMoneyError):
            Money(-50)

    def test_money_addition(self):
        """Сложение денег одной валюты"""
        m1 = Money(100, "USD")
        m2 = Money(50, "USD")
        
        result = m1 + m2
        
        assert result.amount == 150
        assert result.currency == "USD"

    def test_money_addition_different_currencies_fails(self):
        """Нельзя складывать разные валюты"""
        usd = Money(100, "USD")
        eur = Money(100, "EUR")
        
        with pytest.raises(ValueError):
            usd + eur

    def test_money_equality(self):
        """Равенство по значению"""
        m1 = Money(100, "USD")
        m2 = Money(100, "USD")
        
        assert m1 == m2  # Разные объекты, но равны!

    def test_money_immutability(self):
        """Value Object неизменяем (frozen dataclass)"""
        money = Money(100)
        
        with pytest.raises(AttributeError):
            money.amount = 200  # Нельзя изменить!


class TestEmail:
    """Тесты для Value Object Email"""

    @pytest.mark.parametrize("valid_email", [
        "user@example.com",
        "test.user@domain.org",
        "admin@sub.domain.com",
    ])
    def test_valid_emails(self, valid_email):
        """Валидные email адреса"""
        email = Email(valid_email)
        assert str(email) == valid_email

    @pytest.mark.parametrize("invalid_email", [
        "not-an-email",
        "@missing-local.com",
        "missing-at-sign.com",
        "",
        "spaces in@email.com",
    ])
    def test_invalid_emails_raise_error(self, invalid_email):
        """Невалидные email выбрасывают ошибку"""
        with pytest.raises(InvalidEmailError):
            Email(invalid_email)

1.2 Тестирование Entities

# tests/domain/test_entities.py
import pytest
import uuid
from domain.entities import Order
from domain.value_objects import Money, Email
from domain.exceptions import OrderAlreadyPaidError, OrderEmptyError
from domain.events import OrderPaid


class TestOrder:
    """Тесты для Entity Order"""

    @pytest.fixture
    def order(self):
        """Фикстура: создает пустой заказ для каждого теста"""
        return Order(
            order_id=uuid.uuid4(),
            customer=Email("test@example.com")
        )

    @pytest.fixture
    def order_with_items(self, order):
        """Фикстура: заказ с товарами"""
        order.add_item("Book", Money(29.99))
        order.add_item("Pen", Money(5.00))
        return order

    # ═══════════════════════════════════════════════
    # Тесты добавления товаров
    # ═══════════════════════════════════════════════

    def test_add_item_to_order(self, order):
        """Добавление товара в заказ"""
        order.add_item("Python Book", Money(49.99))
        
        assert len(order.items) == 1
        assert order.items[0]["name"] == "Python Book"

    def test_total_price_calculation(self, order_with_items):
        """Расчет общей суммы"""
        total = order_with_items.total_price
        
        assert total.amount == 34.99  # 29.99 + 5.00
        assert total.currency == "USD"

    # ═══════════════════════════════════════════════
    # Тесты оплаты (ИНВАРИАНТЫ)
    # ═══════════════════════════════════════════════

    def test_pay_order_success(self, order_with_items):
        """Успешная оплата заказа"""
        order_with_items.pay()
        
        assert order_with_items.is_paid == True

    def test_pay_empty_order_fails(self, order):
        """ИНВАРИАНТ: Нельзя оплатить пустой заказ"""
        with pytest.raises(OrderEmptyError):
            order.pay()

    def test_pay_twice_fails(self, order_with_items):
        """ИНВАРИАНТ: Нельзя оплатить дважды"""
        order_with_items.pay()
        
        with pytest.raises(OrderAlreadyPaidError):
            order_with_items.pay()

    def test_cannot_add_item_after_payment(self, order_with_items):
        """ИНВАРИАНТ: Нельзя менять оплаченный заказ"""
        order_with_items.pay()
        
        with pytest.raises(OrderAlreadyPaidError):
            order_with_items.add_item("New Item", Money(10))

    # ═══════════════════════════════════════════════
    # Тесты Domain Events
    # ═══════════════════════════════════════════════

    def test_order_paid_event_generated(self, order_with_items):
        """При оплате генерируется событие OrderPaid"""
        order_with_items.pay()
        
        events = order_with_items.collect_events()
        
        assert len(events) == 1
        assert isinstance(events[0], OrderPaid)
        assert events[0].order_id == order_with_items.id

    def test_events_cleared_after_collection(self, order_with_items):
        """События очищаются после сбора"""
        order_with_items.pay()
        order_with_items.collect_events()
        
        assert len(order_with_items.collect_events()) == 0

2. Тестирование Application Layer (Use Cases)

Здесь мы тестируем Use Cases. Они зависят от интерфейсов (Repositories, Services), которые мы мокаем.

Мок (Mock) — это поддельная реализация интерфейса. Она позволяет тестировать логику без реальной БД или email-сервера.

# tests/application/test_use_cases.py
import pytest
from unittest.mock import Mock, MagicMock
import uuid

from application.use_cases import CreateOrderUseCase, PayOrderUseCase
from application.dto import CreateOrderCommand
from application.interfaces import OrderRepository, EmailService, UnitOfWork
from domain.entities import Order
from domain.exceptions import OrderNotFoundError


class TestCreateOrderUseCase:
    """Тесты для CreateOrderUseCase"""

    @pytest.fixture
    def mock_uow(self):
        """Мок Unit of Work с репозиторием"""
        uow = Mock(spec=UnitOfWork)
        uow.orders = Mock(spec=OrderRepository)
        uow.__enter__ = Mock(return_value=uow)
        uow.__exit__ = Mock(return_value=False)
        return uow

    @pytest.fixture
    def mock_email_service(self):
        """Мок email сервиса"""
        return Mock(spec=EmailService)

    @pytest.fixture
    def use_case(self, mock_uow, mock_email_service):
        """Создаем Use Case с моками"""
        return CreateOrderUseCase(
            uow=mock_uow,
            email_service=mock_email_service
        )

    def test_create_order_success(self, use_case, mock_uow, mock_email_service):
        """Успешное создание заказа"""
        # Arrange: готовим входные данные
        command = CreateOrderCommand(
            customer_email="user@example.com",
            items=[{"name": "Book", "price": 29.99}]
        )

        # Act: выполняем
        result = use_case.execute(command)

        # Assert: проверяем результат
        assert result.order_id is not None
        assert result.customer_email == "user@example.com"
        assert result.total_amount == 29.99
        assert result.status == "created"

        # Assert: проверяем, что репозиторий был вызван
        mock_uow.orders.save.assert_called_once()

        # Assert: проверяем, что email был отправлен
        mock_email_service.send_order_confirmation.assert_called_once_with(
            email="user@example.com",
            order_id=result.order_id
        )

    def test_create_order_invalid_email_fails(self, use_case):
        """Невалидный email приводит к ошибке"""
        command = CreateOrderCommand(
            customer_email="not-an-email",  # Невалидный!
            items=[{"name": "Book", "price": 29.99}]
        )

        with pytest.raises(InvalidEmailError):
            use_case.execute(command)

    def test_transaction_rollback_on_error(self, mock_uow, mock_email_service):
        """Транзакция откатывается при ошибке"""
        # Arrange: email сервис выбрасывает ошибку
        mock_email_service.send_order_confirmation.side_effect = Exception("SMTP Error")
        
        use_case = CreateOrderUseCase(
            uow=mock_uow,
            email_service=mock_email_service
        )

        command = CreateOrderCommand(
            customer_email="user@example.com",
            items=[{"name": "Book", "price": 29.99}]
        )

        # Act & Assert
        with pytest.raises(Exception):
            use_case.execute(command)

        # Проверяем, что __exit__ был вызван (значит, транзакция завершена)
        mock_uow.__exit__.assert_called()


class TestPayOrderUseCase:
    """Тесты для PayOrderUseCase"""

    @pytest.fixture
    def existing_order(self):
        """Существующий заказ с товарами"""
        order = Order(uuid.uuid4(), Email("test@example.com"))
        order.add_item("Book", Money(50))
        return order

    def test_pay_existing_order(self, existing_order):
        """Оплата существующего заказа"""
        # Arrange: мокаем репозиторий, который вернет наш заказ
        mock_uow = Mock()
        mock_uow.orders.get_by_id.return_value = existing_order
        mock_uow.__enter__ = Mock(return_value=mock_uow)
        mock_uow.__exit__ = Mock(return_value=False)

        use_case = PayOrderUseCase(uow=mock_uow)

        # Act
        result = use_case.execute(existing_order.id)

        # Assert
        assert result.status == "paid"
        assert existing_order.is_paid == True

    def test_pay_nonexistent_order_fails(self):
        """Оплата несуществующего заказа"""
        mock_uow = Mock()
        mock_uow.orders.get_by_id.return_value = None  # Не найден!
        mock_uow.__enter__ = Mock(return_value=mock_uow)
        mock_uow.__exit__ = Mock(return_value=False)

        use_case = PayOrderUseCase(uow=mock_uow)

        with pytest.raises(OrderNotFoundError):
            use_case.execute(uuid.uuid4())

3. Тестирование Infrastructure Layer (Integration)

Здесь мы тестируем реальные реализации: SQLAlchemy Repository, Email Adapter и т.д. Нужна тестовая БД.

# tests/infrastructure/test_repositories.py
import pytest
import uuid
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

from infrastructure.db.models import Base
from infrastructure.repositories import SqlAlchemyOrderRepository
from domain.entities import Order
from domain.value_objects import Email, Money


# Используем SQLite in-memory для скорости
TEST_DATABASE_URL = "sqlite:///:memory:"


@pytest.fixture(scope="function")
def session():
    """Создает чистую БД для каждого теста"""
    engine = create_engine(TEST_DATABASE_URL)
    Base.metadata.create_all(engine)
    
    Session = sessionmaker(bind=engine)
    session = Session()
    
    yield session
    
    session.close()
    Base.metadata.drop_all(engine)


@pytest.fixture
def repository(session):
    """Репозиторий с тестовой сессией"""
    return SqlAlchemyOrderRepository(session)


class TestSqlAlchemyOrderRepository:
    """Интеграционные тесты репозитория"""

    def test_save_and_retrieve_order(self, repository, session):
        """Сохранение и получение заказа"""
        # Arrange
        order_id = uuid.uuid4()
        order = Order(order_id, Email("test@example.com"))
        order.add_item("Book", Money(29.99))
        order.add_item("Pen", Money(5.00))

        # Act: сохраняем
        repository.save(order)
        session.commit()

        # Act: загружаем
        loaded = repository.get_by_id(order_id)

        # Assert
        assert loaded is not None
        assert loaded.id == order_id
        assert str(loaded.customer) == "test@example.com"
        assert len(loaded.items) == 2
        assert loaded.total_price.amount == 34.99

    def test_get_nonexistent_order_returns_none(self, repository):
        """Несуществующий заказ возвращает None"""
        result = repository.get_by_id(uuid.uuid4())
        assert result is None

    def test_update_existing_order(self, repository, session):
        """Обновление существующего заказа"""
        order_id = uuid.uuid4()
        order = Order(order_id, Email("test@example.com"))
        order.add_item("Book", Money(29.99))
        
        repository.save(order)
        session.commit()

        # Загружаем, изменяем, сохраняем
        loaded = repository.get_by_id(order_id)
        loaded.add_item("Pen", Money(5.00))
        repository.save(loaded)
        session.commit()

        # Проверяем
        reloaded = repository.get_by_id(order_id)
        assert len(reloaded.items) == 2

    def test_order_state_persisted(self, repository, session):
        """Состояние заказа сохраняется (оплачен/не оплачен)"""
        order_id = uuid.uuid4()
        order = Order(order_id, Email("test@example.com"))
        order.add_item("Book", Money(29.99))
        order.pay()

        repository.save(order)
        session.commit()

        loaded = repository.get_by_id(order_id)
        assert loaded.is_paid == True

4. E2E тесты (Presentation Layer)

Проверяем API целиком. Используем TestClient от FastAPI (или Django TestCase).

# tests/e2e/test_api.py
import pytest
from fastapi.testclient import TestClient
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

from main import create_app
from infrastructure.db.models import Base
from presentation.dependencies import get_session


# Переопределяем зависимость БД на тестовую
TEST_DATABASE_URL = "sqlite:///:memory:"


@pytest.fixture(scope="module")
def test_app():
    """Создает тестовое приложение с in-memory БД"""
    engine = create_engine(TEST_DATABASE_URL, connect_args={"check_same_thread": False})
    TestSession = sessionmaker(bind=engine)
    
    Base.metadata.create_all(engine)

    def override_get_session():
        session = TestSession()
        try:
            yield session
        finally:
            session.close()

    app = create_app()
    app.dependency_overrides[get_session] = override_get_session
    
    yield TestClient(app)
    
    Base.metadata.drop_all(engine)


class TestOrdersAPI:
    """E2E тесты API заказов"""

    def test_create_order_success(self, test_app):
        """POST /api/v1/orders - успешное создание"""
        response = test_app.post(
            "/api/v1/orders",
            json={
                "customer_email": "user@example.com",
                "items": [
                    {"name": "Python Book", "price": 49.99, "quantity": 1}
                ]
            }
        )
        
        assert response.status_code == 201
        data = response.json()
        assert "id" in data
        assert data["customer_email"] == "user@example.com"
        assert data["total_amount"] == 49.99
        assert data["status"] == "created"

    def test_create_order_invalid_email(self, test_app):
        """POST /api/v1/orders - невалидный email возвращает 400"""
        response = test_app.post(
            "/api/v1/orders",
            json={
                "customer_email": "not-an-email",
                "items": [{"name": "Book", "price": 10}]
            }
        )
        
        assert response.status_code == 400
        assert "error" in response.json()

    def test_create_order_empty_items(self, test_app):
        """POST /api/v1/orders - пустой список товаров"""
        response = test_app.post(
            "/api/v1/orders",
            json={
                "customer_email": "user@example.com",
                "items": []
            }
        )
        
        assert response.status_code == 422  # Validation Error

    def test_get_order_not_found(self, test_app):
        """GET /api/v1/orders/{id} - несуществующий заказ"""
        fake_id = "00000000-0000-0000-0000-000000000000"
        response = test_app.get(f"/api/v1/orders/{fake_id}")
        
        assert response.status_code == 404

    def test_full_order_flow(self, test_app):
        """Полный цикл: создание → получение → оплата"""
        # 1. Создаем заказ
        create_response = test_app.post(
            "/api/v1/orders",
            json={
                "customer_email": "flow@example.com",
                "items": [{"name": "Book", "price": 100}]
            }
        )
        assert create_response.status_code == 201
        order_id = create_response.json()["id"]

        # 2. Получаем заказ
        get_response = test_app.get(f"/api/v1/orders/{order_id}")
        assert get_response.status_code == 200
        assert get_response.json()["status"] == "created"

        # 3. Оплачиваем
        pay_response = test_app.post(f"/api/v1/orders/{order_id}/pay")
        assert pay_response.status_code == 200
        assert pay_response.json()["status"] == "paid"

        # 4. Повторная оплата должна вернуть ошибку
        pay_again = test_app.post(f"/api/v1/orders/{order_id}/pay")
        assert pay_again.status_code == 409  # Conflict

5. Советы по тестированию

✅ Хорошие практики

  • • Один тест — одна проверка (Arrange-Act-Assert)
  • • Используйте фикстуры для повторяющегося кода
  • • Давайте тестам говорящие имена
  • • Тестируйте граничные случаи и ошибки
  • • Используйте `pytest.mark.parametrize` для множества входных данных
  • • Запускайте unit-тесты на каждый коммит

❌ Антипаттерны

  • • Тесты зависят друг от друга (порядок важен)
  • • Тесты медленные (БД в unit-тестах)
  • • Слишком много моков (тест не проверяет реальное поведение)
  • • Тесты дублируют логику (копипаст из продакшн-кода)
  • • Нет assertion (тест просто "не падает")
  • • Тестирование приватных методов
