    """

        status=result.status,
        created_at=result.created_at
    )

    OrderNotFoundError: (status.HTTP_404_NOT_FOUND, "ORDER_NOT_FOUND"),
    OrderAlreadyPaidError: (status.HTTP_409_CONFLICT, "ORDER_ALREADY_PAID"),

async def domain_error_handler(request: Request, exc: DomainError):
    """
    Global handler –¥–ª—è –≤—Å—ñ—Ö –¥–æ–º–µ–Ω–Ω–∏—Ö –ø–æ–º–∏–ª–æ–∫.
    –ü–µ—Ä–µ—Ç–≤–æ—Ä—é—î DomainError –Ω–∞ –∫—Ä–∞—Å–∏–≤—É JSON –≤—ñ–¥–ø–æ–≤—ñ–¥—å.
    """
    # –®—É–∫–∞—î–º–æ –º–∞–ø—ñ–Ω–≥ –¥–ª—è –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ —Ç–∏–ø—É –ø–æ–º–∏–ª–∫–∏
    error_type = type(exc)
    status_code, error_code = ERROR_MAPPING.get(
        error_type, 
        (status.HTTP_400_BAD_REQUEST, "DOMAIN_ERROR")
    )
    
    return JSONResponse(
        status_code=status_code,
        content={
            "error": error_type.__name__,
            "detail": str(exc),
            "code": error_code
        }
    )
