# infrastructure/repositories.py
class SqlAlchemyOrderRepository(OrderRepository):
    def __init__(self, session):
        self.session = session

    def add(self, order: Order):
        # –ú–∞–ø—ñ–Ω–≥ Entity -> DB Model –≤—ñ–¥–±—É–≤–∞—î—Ç—å—Å—è —Ç—É—Ç
        model = OrderModel.from_entity(order)
        self.session.add(model)

    def get(self, order_id: str) -> Optional[Order]:
        model = self.session.query(OrderModel).filter_by(id=order_id).first()
        return model.to_entity() if model else None

# infrastructure/uow.py
from sqlalchemy.orm import Session

class SqlAlchemyUnitOfWork(UnitOfWork):
    def __init__(self, session_factory):
        self.session_factory = session_factory

    def __enter__(self):
        self.session = self.session_factory()
        self.orders = SqlAlchemyOrderRepository(self.session) # –Ü–Ω'—î–∫—Ü—ñ—è —Å–µ—Å—ñ—ó
        return self

    def __exit__(self, *args):
        super().__exit__(*args)
        self.session.close()

    def commit(self):
        self.session.commit()

    def rollback(self):
        self.session.rollback()
