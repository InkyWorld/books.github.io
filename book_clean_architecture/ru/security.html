class ContextIdentityProvider(IIdentityProvider):
    def get_current_user_id(self) -> str:
        uid = _current_user_id.get()
        if not uid:
            raise UnauthorizedError()
        return uid

async def auth_middleware(request: Request, call_next):
    token = request.headers.get("Authorization")
    user_id = decode_token(token)
    token = _current_user_id.set(user_id) # Set context
    try:
        response = await call_next(request)
    finally:
        _current_user_id.reset(token) # Clean up

class CancelOrderUseCase:
    def __init__(self, identity: IIdentityProvider, repo: OrderRepo):
        self.identity = identity
        
    def execute(self, order_id: str):
        user_id = self.identity.get_current_user_id()
        order = self.repo.get(order_id)
        
        if order.owner_id != user_id and "ADMIN" not in self.identity.get_permissions():
            raise PermissionDenied("Not yours!")
            
        order.cancel()
