# RUNNER STAGE

WORKDIR /app
COPY --from=builder /app/wheels /wheels
COPY --from=builder /app/requirements.txt .
RUN pip install --no-cache /wheels/*

COPY src/ src/
RUN adduser --disabled-password appuser
USER appuser

CMD ["python", "-m", "src.main"]
