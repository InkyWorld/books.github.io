Repository & Unit of Work

Abstraction over database and transaction management.

Repository

A repository creates the illusion that all domain objects are stored in an in-memory collection (Collection-oriented). It hides the details of working with the database (SQL, ORM).

class AbstractRepository(ABC):
    @abstractmethod
    def add(self, aggregate):
        pass

    @abstractmethod
    def get(self, id) -> Aggregate:
        pass

# Implementation with SQLAlchemy
class SqlAlchemyRepository(AbstractRepository):
    def __init__(self, session):
        self.session = session

    def add(self, order):
        self.session.add(order)

    def get(self, order_id):
        return self.session.query(Order).filter_by(id=order_id).one()

Unit of Work

The Unit of Work pattern tracks all changes made during a business transaction and saves them atomically (Atomic Commit). This solves the problem of partial data updates.

# Using in Application Layer
def handle_order(cmd):
    with uow: # Start Transaction
        order = uow.orders.get(cmd.order_id)
        order.approve()
        uow.commit() # Commit or Rollback on error

Example of Unit of Work Implementation

class AbstractUnitOfWork(ABC):
    orders: AbstractRepository

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self.rollback()

    @abstractmethod
    def commit(self):
        raise NotImplementedError

    @abstractmethod
    def rollback(self):
        raise NotImplementedError

class SqlAlchemyUnitOfWork(AbstractUnitOfWork):
    def __init__(self, session_factory):
        self.session_factory = session_factory

    def __enter__(self):
        self.session = self.session_factory()
        self.orders = SqlAlchemyRepository(self.session)
        return self

    def __exit__(self, exc_type, *args):
        super().__exit__(exc_type, *args)
        self.session.close()

    def commit(self):
        self.session.commit()

    def rollback(self):
        self.session.rollback()
