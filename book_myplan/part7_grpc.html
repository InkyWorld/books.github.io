package scaletrade.orders;

import "google/protobuf/timestamp.proto";

// Enums
enum OrderSide {
  BUY = 0;
  SELL = 1;
}

  PENDING = 0;

  CANCELLED = 3;
}

// Messages
message Order {
  int64 id = 1;
  int64 user_id = 2;
  string symbol = 3;
  OrderSide side = 4;
  string quantity = 5;  // Decimal as string
  string price = 6;
  string filled_quantity = 7;
  OrderStatus status = 8;
  google.protobuf.Timestamp created_at = 9;
  google.protobuf.Timestamp updated_at = 10;
}

  int64 user_id = 1;
  string symbol = 2;
  OrderSide side = 3;
  string quantity = 4;
  string price = 5;
}

  Order order = 1;
  repeated string errors = 2;
}

  int64 order_id = 1;
}

message ListOrdersRequest {
  int64 user_id = 1;
  string symbol = 2;
  OrderStatus status = 3;
  int32 page_size = 4;
  string page_token = 5;
}

message ListOrdersResponse {
  repeated Order orders = 1;
  string next_page_token = 2;
  int32 total_count = 3;
}

}

# interface/grpc/order_servicer.py
import grpc
from decimal import Decimal
from datetime import datetime
from google.protobuf.timestamp_pb2 import Timestamp

from generated import order_service_pb2, order_service_pb2_grpc
from application.order_service import OrderService
from domain.entities.order import Order as OrderEntity

class OrderServicer(order_service_pb2_grpc.OrderServiceServicer):
    def __init__(self, order_service: OrderService):
        self.order_service = order_service
    
    async def CreateOrder(
        self,
        request: order_service_pb2.CreateOrderRequest,
        context: grpc.aio.ServicerContext
    ) -> order_service_pb2.CreateOrderResponse:
        """–°–æ–∑–¥–∞–Ω–∏–µ –æ—Ä–¥–µ—Ä–∞ (Unary RPC)"""
        try:
            order_entity = await self.order_service.create_order(
                user_id=request.user_id,
                symbol=request.symbol,
                side=request.side.name,
                quantity=Decimal(request.quantity),
                price=Decimal(request.price)
            )
            
            return order_service_pb2.CreateOrderResponse(
                order=self._entity_to_proto(order_entity)
            )
        except ValueError as e:
            return order_service_pb2.CreateOrderResponse(
                errors=[str(e)]
            )
    
    async def GetOrder(
        self,
        request: order_service_pb2.GetOrderRequest,
        context: grpc.aio.ServicerContext
    ) -> order_service_pb2.Order:
        """–ü–æ–ª—É—á–µ–Ω–∏–µ –æ—Ä–¥–µ—Ä–∞ –ø–æ ID"""
        order = await self.order_service.get_by_id(request.order_id)
        
        if not order:
            context.set_code(grpc.StatusCode.NOT_FOUND)
            context.set_details(f"Order {request.order_id} not found")
            return order_service_pb2.Order()
        
        return self._entity_to_proto(order)
    
    async def WatchOrders(
        self,
        request: order_service_pb2.GetOrderRequest,
        context: grpc.aio.ServicerContext
    ):
        """Server streaming - –ø–æ–¥–ø–∏—Å–∫–∞ –Ω–∞ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è –æ—Ä–¥–µ—Ä–∞"""
        async for update in self.order_service.subscribe_to_order(request.order_id):
            if context.cancelled():
                break
            
            yield order_service_pb2.OrderUpdate(
                order=self._entity_to_proto(update.order),
                update_type=update.type
            )
    
    async def BatchCreateOrders(
        self,
        request_iterator,
        context: grpc.aio.ServicerContext
    ) -> order_service_pb2.CreateOrderResponse:
        """Client streaming - –ø–∞–∫–µ—Ç–Ω–æ–µ —Å–æ–∑–¥–∞–Ω–∏–µ"""
        created_orders = []
        errors = []
        
        async for request in request_iterator:
            try:
                order = await self.order_service.create_order(
                    user_id=request.user_id,
                    symbol=request.symbol,
                    side=request.side.name,
                    quantity=Decimal(request.quantity),
                    price=Decimal(request.price)
                )
                created_orders.append(order)
            except ValueError as e:
                errors.append(f"{request.symbol}: {e}")
        
        return order_service_pb2.CreateOrderResponse(
            order=self._entity_to_proto(created_orders[0]) if created_orders else None,
            errors=errors
        )
    
    def _entity_to_proto(self, entity: OrderEntity) -> order_service_pb2.Order:
        """–ö–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è Entity -> Protobuf"""
        created_ts = Timestamp()
        created_ts.FromDatetime(entity.created_at)
        
        updated_ts = Timestamp()
        updated_ts.FromDatetime(entity.updated_at)
        
        return order_service_pb2.Order(
            id=entity.id,
            user_id=entity.user_id,
            symbol=entity.symbol,
            side=order_service_pb2.OrderSide.Value(entity.side),
            quantity=str(entity.quantity),
            price=str(entity.price),
            filled_quantity=str(entity.filled_quantity),
            status=order_service_pb2.OrderStatus.Value(entity.status),
            created_at=created_ts,
            updated_at=updated_ts
        )

import asyncio
import grpc
from grpc_reflection.v1alpha import reflection

from generated import order_service_pb2_grpc

from infrastructure.di import setup_di

async def serve():
    """–ó–∞–ø—É—Å–∫ gRPC —Å–µ—Ä–≤–µ—Ä–∞"""
    container = setup_di()
    order_service = await container.get(OrderService)
    
    server = grpc.aio.server()
    order_service_pb2_grpc.add_OrderServiceServicer_to_server(
        OrderServicer(order_service),
        server
    )
    
    # –í–∫–ª—é—á–µ–Ω–∏–µ reflection –¥–ª—è grpcurl/grpcui
    SERVICE_NAMES = (
        order_service_pb2.DESCRIPTOR.services_by_name['OrderService'].full_name,
        reflection.SERVICE_NAME,
    )
    reflection.enable_server_reflection(SERVICE_NAMES, server)
    
    # TLS –¥–ª—è production
    with open('certs/server.key', 'rb') as f:
        private_key = f.read()
    with open('certs/server.crt', 'rb') as f:
        certificate_chain = f.read()
    
    server_credentials = grpc.ssl_server_credentials(
        [(private_key, certificate_chain)]
    )
    
    server.add_secure_port('[::]:50051', server_credentials)
    await server.start()
    
    print("gRPC server started on port 50051")
    await server.wait_for_termination()

if __name__ == '__main__':
    asyncio.run(serve())

# interface/grpc/interceptors.py
import grpc
import time
import structlog
from typing import Callable, Any

logger = structlog.get_logger()

class LoggingInterceptor(grpc.aio.ServerInterceptor):
    """–õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –≤—Å–µ—Ö gRPC –≤—ã–∑–æ–≤–æ–≤"""
    
    async def intercept_service(
        self,
        continuation: Callable,
        handler_call_details: grpc.HandlerCallDetails
    ) -> grpc.RpcMethodHandler:
        start = time.time()
        
        method = handler_call_details.method
        logger.info("grpc_call_start", method=method)
        
        try:
            response = await continuation(handler_call_details)
            duration = time.time() - start
            logger.info("grpc_call_success", method=method, duration=duration)
            return response
        except Exception as e:
            duration = time.time() - start
            logger.error("grpc_call_error", method=method, duration=duration, error=str(e))
            raise

class AuthInterceptor(grpc.aio.ServerInterceptor):
    """–ü—Ä–æ–≤–µ—Ä–∫–∞ JWT —Ç–æ–∫–µ–Ω–∞"""
    
    async def intercept_service(
        self,
        continuation: Callable,
        handler_call_details: grpc.HandlerCallDetails
    ) -> grpc.RpcMethodHandler:
        metadata = dict(handler_call_details.invocation_metadata)
        
        token = metadata.get("authorization", "").replace("Bearer ", "")
        
        if not token:
            raise grpc.RpcError(
                grpc.StatusCode.UNAUTHENTICATED,
                "Missing authorization token"
            )
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ JWT
        try:
            user_id = verify_jwt(token)
            # –î–æ–±–∞–≤–ª—è–µ–º user_id –≤ –∫–æ–Ω—Ç–µ–∫—Å—Ç
            handler_call_details.invocation_metadata.append(
                ("user-id", str(user_id))
            )
        except ValueError:
            raise grpc.RpcError(
                grpc.StatusCode.UNAUTHENTICATED,
                "Invalid token"
            )
        
        return await continuation(handler_call_details)
